00023 double CubicPath(double startPoint, double endPoint, clock_t time, clock_t timePeriod)
00024 {
00025         time /= CLOCKS_PER_SEC;
00026         timePeriod /= CLOCKS_PER_SEC;
00027         // Taken from the Robotics Trajectory Planning Notes
00028         auto a0 = startPoint;
00029         auto a2 = 3*(endPoint - startPoint)/(timePeriod*timePeriod);
00030         auto a3 = -2 * (endPoint - startPoint)/(timePeriod*timePeriod*timePeriod);
00031 
00032         return a0 + a2*time*time + a3*time*time*time;
